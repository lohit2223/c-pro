#include <stdio.h>

int main() {
    char str[] = "Remove all spaces from this string.";
    char result[100];  // Make sure this is large enough
    int i = 0, j = 0;

    while (str[i]) {
        if (str[i] != ' ') {
            result[j++] = str[i];
        }
        i++;
    }
    result[j] = '\0';  // Null-terminate the result string

    printf("Original: \"%s\"\n", str);
    printf("Without spaces: \"%s\"\n", result);

    return 0;
}
