#include <stdio.h>

int main() {
    char str[100];
    int i, length = 0;

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);  // read string with spaces

    // Find length of string
    for (i = 0; str[i] != '\0'; i++) {
        length++;
    }

    // fgets() also stores '\n' before '\0', so adjust length if needed
    if (str[length - 1] == '\n') {
        length--; 
    }

    printf("Length of the string is: %d\n", length);

    return 0;
}
