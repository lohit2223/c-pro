#include <stdio.h>
#include <math.h>

int main() {
    int n, temp, r, sum = 0, digits = 0;
    printf("Enter number: ");
    scanf("%d", &n);

    temp = n;
    while (temp) { digits++; temp /= 10; }

    temp = n;
    while (temp) {
        r = temp % 10;
        sum += pow(r, digits);
        temp /= 10;
    }

    if (sum == n) printf("%d is Armstrong\n", n);
    else printf("%d is not Armstrong\n", n);

    return 0;
}

OUPUT:
Enter number: 5
5 is Armstrong
