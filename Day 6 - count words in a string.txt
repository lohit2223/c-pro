#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main() {
    char str[200];
    int i, words = 0;
    int inWord = 0; // Flag to track if we are inside a word

    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin); // read string with spaces

    for (i = 0; str[i] != '\0'; i++) {
        if (!isspace(str[i]) && inWord == 0) {
            // Start of a new word
            inWord = 1;
            words++;
        } else if (isspace(str[i])) {
            // End of a word
            inWord = 0;
        }
    }

    printf("Number of words: %d\n", words);

    return 0;
}
