#include <stdio.h>

int main() {
    char str[] = "programming in C is fun";
    char target = 'g';  // character to count
    int count = 0;

    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == target) {
            count++;
        }
    }

    printf("Character '%c' appears %d times in \"%s\"\n", target, count, str);

    return 0;
}
