#include <stdio.h>
#include <string.h>
int main() {
    char name[50], department[50];
    char subjects[5][30];
    int semester;
    int marks[5], i, total = 0, pass = 1;
    float percentage;
    printf("Enter Student Name: ");
    scanf(" %[^\n]", name);
    printf("Enter Department: ");
    scanf(" %[^\n]", department);
    printf("Enter Semester: ");
    scanf("%d", &semester);
    printf("\nEnter Subject Names and Marks:\n");
    for (i = 0; i < 5; i++) {
        printf("Subject %d Name: ", i + 1);
        scanf(" %[^\n]", subjects[i]);
        printf("Marks for %s: ", subjects[i]);
        scanf("%d", &marks[i]);
        total += marks[i];
        if (marks[i] < 40) {
            pass = 0;
        }
    }
    percentage = (float)total / 5;
    printf("\n--- Student Report ---\n");
    printf("Name       : %s\n", name);
    printf("Department : %s\n", department);
    printf("Semester   : %d\n", semester);
    printf("\nSubject-wise Marks:\n");
    for (i = 0; i < 5; i++) {
        printf("%-15s : %d\n", subjects[i], marks[i]);
    }
    printf("\nPass/Arrear : %s\n", (pass) ? "Pass" : "Arrear");
    printf("Percentage  : %.2f%%\n", percentage);
    return 0;
}
