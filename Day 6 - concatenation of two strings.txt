#include <stdio.h>

int main() {
    char str1[100], str2[100];
    int i, j;

    // Input two strings
    printf("Enter first string: ");
    fgets(str1, sizeof(str1), stdin);

    printf("Enter second string: ");
    fgets(str2, sizeof(str2), stdin);

    // Remove newline character if present in str1
    for (i = 0; str1[i] != '\0'; i++) {
        if (str1[i] == '\n') {
            str1[i] = '\0';
            break;
        }
    }
    // Remove newline character if present in str2
    for (i = 0; str2[i] != '\0'; i++) {
        if (str2[i] == '\n') {
            str2[i] = '\0';
            break;
        }
    }

    // Find the end of str1
    for (i = 0; str1[i] != '\0'; i++);

    // Copy str2 into str1
    for (j = 0; str2[j] != '\0'; j++, i++) {
        str1[i] = str2[j];
    }

    // Null terminate the concatenated string
    str1[i] = '\0';

    // Output concatenated string
    printf("Concatenated string: %s\n", str1);

    return 0;
}
